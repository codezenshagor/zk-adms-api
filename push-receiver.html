03 Payload Format (PushDispatcher)

ZK PushDispatcher এই format এ JSON POST করে receiver endpoint এ।

JSON — PushDispatcher Payload 
{
  "api_key": "84614e930f36953cb6e8d39b94672bc9b8a466ba092acfd4",
  "records": [
    {
      "sn":           "NYU7251601368",   // device serial number
      "pin":          "1001",            // user employee ID / PIN
      "log_datetime": "2026-03-31 09:00:00",
      "status":       0,                 // 0=check-in, 1=check-out, 4=OT-in, 5=OT-out
      "source":       "attlog"           // attlog / alarm / access
    },
    {
      "sn":           "NYU7251601368",
      "pin":          "1002",
      "log_datetime": "2026-03-31 09:05:22",
      "status":       0,
      "source":       "attlog"
    }
  ]
}
⚠️ PIN 0 ফিল্টার: Receiver automatically PIN = "0" বা empty PIN records skip করে। এগুলো device এর system punch — attendance log হিসেবে count হয় না।
04 SQLite Schema

Lightweight deployment এর জন্য SQLite। Small-medium office এর জন্য ideal।

TABLE device Registered ZK devices
SQLite 
CREATE TABLE IF NOT EXISTS device (
  id               INTEGER  PRIMARY KEY AUTOINCREMENT,
  name             TEXT     NOT NULL  DEFAULT '',
  serial           TEXT     NOT NULL  UNIQUE,
  api_key          TEXT     DEFAULT  '',
  active_status    INTEGER  DEFAULT  1,
  last_connection  TEXT     DEFAULT  NULL,
  created_at       TEXT     DEFAULT  (datetime('now', 'localtime'))
);
TABLE att_log Attendance punch records
SQLite 
CREATE TABLE IF NOT EXISTS att_log (
  id             INTEGER  PRIMARY KEY AUTOINCREMENT,
  main_id        TEXT     NOT NULL,               -- user PIN / employee ID
  state          TEXT     DEFAULT '0',             -- 0=in,1=out,4=OT-in,5=OT-out
  timestamp      TEXT     NOT NULL,               -- Y-m-d H:i:s
  type           TEXT     DEFAULT 'attlog',        -- attlog / access / alarm
  device_serial  TEXT     DEFAULT NULL,
  UNIQUE (main_id, timestamp, device_serial)  -- prevent duplicate punches
);

CREATE INDEX IF NOT EXISTS idx_att_main_id    ON att_log(main_id);
CREATE INDEX IF NOT EXISTS idx_att_timestamp  ON att_log(timestamp);
CREATE INDEX IF NOT EXISTS idx_att_device     ON att_log(device_serial);

PHP — Laravel Route + Controller 
// routes/api.php
Route::post('/zkrcv2', [ZkReceiverController::class, 'receive']);

// ─────────────────────────────────────────────────
// app/Http/Controllers/ZkReceiverController.php
// ─────────────────────────────────────────────────
namespace App\Http\Controllers;

use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;
use Carbon\Carbon;

class ZkReceiverController extends Controller {

  private const API_KEY = '84614e930f36953cb6e8d39b94672bc9b8a466ba092acfd4';

  public function receive(Request $request) {
    // Validate API key
    $key = $request->header('X-Api-Key')
          ?? $request->input('api_key', '');
    if ($key !== self::API_KEY)
      return response()->json(['status'=>'error'], 401);

    $records = $request->input('records', []);
    if (empty($records))
      return response()->json(['status'=>'ok']);

    $sn = $records[0]['sn'] ?? '';
    $inserted = 0;

    DB::transaction(function () use ($records, $sn, &$inserted) {
      foreach ($records as $r) {
        $pin = trim($r['pin'] ?? '');
        if (!$pin || $pin === '0') continue;

        DB::table('att_log')->insertOrIgnore([
          'main_id'       => $pin,
          'state'         => (string)($r['status'] ?? 0),
          'timestamp'     => Carbon::parse($r['log_datetime']),
          'type'          => $r['source'] ?? 'attlog',
          'device_serial' => $sn,
        ]);
        $inserted++;
      }
    });

    return response()->json(['status'=>'ok', 'inserted'=>$inserted]);
  }
}
08 Node.js Setup

Express.js + better-sqlite3 দিয়ে receiver।

Node.js — Express receiver 
// npm install express better-sqlite3
const express  = require('express');
const Database = require('better-sqlite3');

const app       = express();
const db        = new Database('./attendance.db');
const API_KEY   = '84614e930f36953cb6e8d39b94672bc9b8a466ba092acfd4';

app.use(express.json());

// Setup DB
db.exec(`
  CREATE TABLE IF NOT EXISTS att_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    main_id TEXT NOT NULL, state TEXT DEFAULT '0',
    timestamp TEXT NOT NULL, type TEXT DEFAULT 'attlog',
    device_serial TEXT,
    UNIQUE(main_id, timestamp, device_serial)
  )
`);

const insert = db.prepare(`
  INSERT OR IGNORE INTO att_log (main_id,state,timestamp,type,device_serial)
  VALUES (?,?,?,?,?)
`);

app.post('/zkrcv2', (req, res) => {
  const key = req.headers['x-api-key'] || req.body?.api_key;
  if (key !== API_KEY) return res.status(401).json({ status: 'error' });

  const { records = [] } = req.body;
  if (!records.length) return res.json({ status: 'ok' });

  const sn = records[0].sn;
  let inserted = 0;

  const run = db.transaction(() => {
    for (const r of records) {
      if (!r.pin || r.pin === '0') continue;
      const res = insert.run(r.pin, String(r.status||0), r.log_datetime, r.source||'attlog', sn);
      inserted += res.changes;
    }
  });
  run();
  res.json({ status: 'ok', inserted });
});

app.listen(3000, () => console.log('Receiver running on :3000'));
